1"""In-process wrapper around the `yledl` Python API. 

2 

3We use the upstream Python package rather than shelling out to its CLI. The 

4public surface in `yledl.__all__` is enough to drive a download, but 

5constructing the downloader requires three helpers that live in submodules 

6(`yledl.http.HttpClient`, `yledl.titleformatter.TitleFormatter`, 

7`yledl.geolocation.AreenaGeoLocation`) — the upstream CLI entry point 

8(`yledl/yledl.py:main`) uses these same imports, so we accept the same 

9internal-API surface yle-dl itself depends on. 

10""" 

11 

12from __future__ import annotations 

13 

14from dataclasses import dataclass 

15from pathlib import Path 

16from typing import Any 

17 

18from yledl import RD_FAILED, RD_INCOMPLETE, RD_SUCCESS, IOContext, StreamFilters, YleDlDownloader 

19from yledl.geolocation import AreenaGeoLocation 

20from yledl.http import HttpClient 

21from yledl.titleformatter import TitleFormatter 

22 

23# Mirrors the Bash OUTPUT_TEMPLATE — yle-dl interpolates these tokens. 

24OUTPUT_TEMPLATE = "${series}/${season}/${series} - ${episode_or_date} - ${title}" 

25 

26 

27@dataclass(frozen=True, slots=True) 

28class Episode: 

29 """Subset of the yle-dl metadata schema (docs/metadata.md upstream) we use.""" 

30 

31 filename: str 

32 title: str 

33 description: str 

34 thumbnail: str 

35 duration_seconds: int 

36 publish_timestamp: str 

37 webpage: str 

38 program_id: str 

39 

40 @classmethod 

41 def from_metadata(cls, item: dict[str, Any]) -> Episode: 

42 return cls( 

43 filename=item.get("filename", ""), 

44 title=item.get("title", ""), 

45 description=item.get("description", ""), 

46 thumbnail=item.get("thumbnail", ""), 

47 duration_seconds=int(item.get("duration_seconds") or 0), 

48 publish_timestamp=item.get("publish_timestamp", ""), 

49 webpage=item.get("webpage", ""), 

50 program_id=item.get("program_id", ""), 

51 ) 

52 

53 

54@dataclass(frozen=True, slots=True) 

55class _Wiring: 

56 downloader: YleDlDownloader 

57 io: IOContext 

58 filters: StreamFilters 

59 

60 

61def _build_wiring(destdir: Path) -> _Wiring: 

62 # `preferred_format` and `resume` are left at None/False by IOContext 

63 # but the yle-dl CLI always sets them (defaults: 'mkv', True). Without 

64 # `preferred_format`, `Downloader.file_extension()` crashes during a 

65 # real download. With `resume=True`, re-runs skip already-downloaded 

66 # files instead of refusing to overwrite. 

67 io = IOContext( 

68 destdir=str(destdir), 

69 create_dirs=True, 

70 preferred_format="mkv", 

71 resume=True, 

72 ) 

73 httpclient = HttpClient(io) 

74 title_formatter = TitleFormatter(OUTPUT_TEMPLATE) 

75 geolocation = AreenaGeoLocation(httpclient) 

76 downloader = YleDlDownloader(geolocation, title_formatter, httpclient) 

77 filters = StreamFilters() 

78 return _Wiring(downloader=downloader, io=io, filters=filters) 

79 

80 

81def fetch_episode_metadata(url: str, destdir: Path) -> list[Episode]: 

82 """Return one Episode per stream detected at `url`. 

83 

84 Mirrors `yle-dl --showmetadata` but skips the JSON-on-stdout round trip. 

85 """ 

86 w = _build_wiring(destdir) 

87 raw = w.downloader.get_metadata(url, w.io, latest_only=False) 

88 return [Episode.from_metadata(item) for item in raw] 

89 

90 

91class DownloadFailed(RuntimeError): 

92 """yle-dl returned RD_FAILED.""" 

93 

94 

95def download_clips(url: str, destdir: Path) -> int: 

96 """Download every clip discovered at `url`. Returns the yle-dl exit code. 

97 

98 Raises DownloadFailed when the result is RD_FAILED. RD_INCOMPLETE is 

99 returned as-is so the caller can warn instead of aborting. 

100 """ 

101 w = _build_wiring(destdir) 

102 code: int = int(w.downloader.download_clips(url, w.io, w.filters)) 

103 if code == RD_FAILED: 

104 raise DownloadFailed(f"yle-dl reported a failure downloading {url!r}") 

105 return code
